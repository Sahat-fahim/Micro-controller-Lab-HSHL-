void setup()
{
  pinMode(4, INPUT);
  pinMode(13, OUTPUT);
  pinMode(2, INPUT);
  pinMode(12, OUTPUT);
}

void loop()
{
  if (digitalRead(4) == HIGH) {
    digitalWrite(13, HIGH);
    printf("going up");
  } else {
    digitalWrite(13, LOW);
  }
  if (digitalRead(2) == HIGH) {
    digitalWrite(12, HIGH);
  } else {
    digitalWrite(12, LOW);
  }
  delay(10); // Delay a little bit to improve simulation performance
}
